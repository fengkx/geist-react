<Attributes edit="/pages/en-us/components/window.mdx">
<Attributes.Title>Window.Props</Attributes.Title>

| Attribute        | Description               | Type                              | Accepted values | Default |
| ---------------- | ------------------------- | --------------------------------- | --------------- | ------- |
| **width**        | window width              | `number` / `string`               | -               | `100%`  |
| **height**       | window height             | `number` / `string`               | -               | `240`   |
| **headerHeight** | window header height      | `number` / `string`               | -               | `40`    |
| **title**        | window title text         | `string`                          | -               | -       |
| **url**          | window url text           | `string`                          | -               | -       |
| **tabs**         | window tabs               | `{title: string, value:string}[]` | -               | -       |
| **selected**     | window selected tab value | `string`                          | -               | -       |
| ...              | native props              | `HTMLAttributes`                  | -               | -       |

</Attributes>

export default ({ children }) => <Layout meta={meta}>{children}</Layout>
